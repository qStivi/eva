Memory Architecture

class TwoTrackContextManager:
    def __init__(self):
        self.conversation_track = []  # Clean conversation history
        self.context_engine = ContextEngine()  # Dynamic side context
        
    async def build_prompt(self, text_input: str, user_id: str) -> dict:
        # Track 1: Get recent conversation (just text, no artifacts)
        conversation = self.get_conversation_window(user_id)
        
        # Track 2: Retrieve relevant context (never shown to user)
        context = await self.context_engine.retrieve(
            query=text_input,
            sources={
                "vector_memories": await self.search_memories(text_input),
                "temporal_context": self.get_temporal_context(),
                "user_patterns": await self.get_user_patterns(user_id),
                "logseq_notes": await self.search_logseq(text_input),
                "character_state": self.character.get_current_state()
            }
        )

class DesktopAwareness:
    async def continuous_awareness(self):
        screenshot = capture_full_screen()
        analysis = await self.vision_model.analyze(screenshot)
        
        context = {
            "applications": analysis.open_windows,
            "content": analysis.readable_text,
            "activity_pattern": self.activity_memory.classify(),
            "time_on_task": self.activity_memory.current_task_duration()
        }
        
        # Character forms opinions and maybe comments
        if context["time_on_task"] > hours(3):
            if self.character.cares_about_health:
                return "You've been at this for hours... maybe stretch?"
                
        if "game" in context["applications"]:
            if self.character.likes_games:
                return "Ooh what are you playing?"

# FastAPI server with WebSocket support
@app.websocket("/ws/conversation")
async def conversation_websocket(websocket: WebSocket):
    await websocket.accept()
    
    while True:
        data = await websocket.receive_json()
        
        if data["type"] == "text":
            # Direct text processing
            response = await character.process_message(
                data["content"], 
                data["user_id"]
            )
            
        elif data["type"] == "audio":
            # Audio -> Text -> Response -> Audio
            text = await stt.transcribe(data["content"])
            response = await character.process_message(text, data["user_id"])
            audio = await tts.synthesize(response, character.voice)
            
# Multi-platform endpoint
@app.post("/api/platform/message")
async def platform_message(request: PlatformRequest):
    response = await character.handle_platform_message(
        platform=request.platform,
        message=request.message,
        context=request.platform_context
    )
    return response
